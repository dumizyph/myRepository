import React, { useEffect } from "react";

import { Box, Typography } from "@mui/material";

import { motion, useAnimation } from "framer-motion";

import { useInView } from "react-intersection-observer";

import SpeedIcon from '@mui/icons-material/Speed';

import DevicesIcon from '@mui/icons-material/Devices';

import SearchIcon from '@mui/icons-material/Search';

import SecurityIcon from '@mui/icons-material/Security';

import AssessmentIcon from '@mui/icons-material/Assessment';

import DesignServicesIcon from '@mui/icons-material/DesignServices';


// Data for industries with icons

const industries = [

  {

    icon: <SpeedIcon style={{ fontSize: '3rem', color: '#ffffff' }} />,

    title: "Technology and IT Services",

    description: `Custom software, cloud infrastructure, and cybersecurity services.`,

  },

  {

    icon: <DevicesIcon style={{ fontSize: '3rem', color: '#ffffff' }} />,

    title: "Healthcare and Life Sciences",

    description: `Improving patient care and streamlining medical processes.`,

  },

  {

    icon: <SearchIcon style={{ fontSize: '3rem', color: '#ffffff' }} />,

    title: "Finance and Banking",

    description: `Secure payment processing, online banking, and financial tools.`,

  },

  {

    icon: <SecurityIcon style={{ fontSize: '3rem', color: '#ffffff' }} />,

    title: "Retail and E-Commerce",

    description: `E-commerce platforms, personalized shopping, and inventory management.`,

  },

  {

    icon: <AssessmentIcon style={{ fontSize: '3rem', color: '#ffffff' }} />,

    title: "Manufacturing and Supply Chain",

    description: `Optimizing manufacturing processes and supply chain management.`,

  },

  {

    icon: <DesignServicesIcon style={{ fontSize: '3rem', color: '#ffffff' }} />,

    title: "Education and E-Learning",

    description: `Building online platforms and interactive learning tools.`,

  },

];


// Motion Variants

const flipVariants = {

  initial: { rotateY: 0 },

  hover: { rotateY: 180, transition: { duration: 0.6 } },

};


const revealVariants = {

  hidden: { opacity: 0, y: 50 },

  visible: { opacity: 1, y: 0, transition: { duration: 0.8 } },

};


// 3D Flip Cards Component

const FlipIndustriesSection = () => {

  const [ref, inView] = useInView({ triggerOnce: true });

  const controls = useAnimation();


  useEffect(() => {

    if (inView) {

      controls.start("visible");

    }

  }, [controls, inView]);


  return (

    <Box

      sx={{

        backgroundColor: "#111", // Dark background to make the design pop

        padding: { xs: "40px 20px", md: "80px 40px" },

        width: "100%",

        position: "relative",

        overflow: "hidden",

      }}

    >

      <Typography

        variant="h3"

        sx={{

          color: "#ffffff", // White title for contrast

          marginBottom: "40px",

          fontSize: { xs: "1.8rem", md: "3.5rem" },

          textAlign: "center",

          fontWeight: "bold",

        }}

      >

        Industries We Serve

      </Typography>


      {/* Grid Layout for the Flip Cards */}

      <Box

        sx={{

          display: "grid",

          gridTemplateColumns: {

            xs: "1fr",

            sm: "1fr 1fr",

            md: "repeat(3, 1fr)", // 3 columns on larger screens

          },

          gap: "50px",

          maxWidth: "1200px",

          margin: "0 auto",

        }}

      >

        {industries.map((industry, index) => (

          <Box

            key={index}

            sx={{

              perspective: "1000px", // Perspective for 3D flip effect

              position: "relative",

            }}

          >

            <motion.div

              variants={flipVariants}

              initial="initial"

              whileHover="hover"

              style={{

                width: "100%",

                height: "300px",

                borderRadius: "20px",

                position: "relative",

                transformStyle: "preserve-3d", // Preserve 3D flip effect

                transition: "transform 0.6s", // Smooth flip transition

              }}

            >

              {/* Front Side of the Card */}

              <motion.div

                style={{

                  width: "100%",

                  height: "100%",

                  backgroundColor: "#1e1e1e", // Dark front card background

                  borderRadius: "20px",

                  display: "flex",

                  justifyContent: "center",

                  alignItems: "center",

                  boxShadow: "0 8px 20px rgba(0, 0, 0, 0.3)", // Subtle shadow

                  position: "absolute",

                  backfaceVisibility: "hidden", // Hide back when front is visible

                  transform: "rotateY(0deg)", // Default position

                }}

              >

                {industry.icon}

              </motion.div>


              {/* Back Side of the Card */}

              <motion.div

                style={{

                  width: "100%",

                  height: "100%",

                  backgroundColor: "#0077ff", // Blue background for the back

                  borderRadius: "20px",

                  display: "flex",

                  flexDirection: "column",

                  justifyContent: "center",

                  alignItems: "center",

                  padding: "20px",

                  textAlign: "center",

                  color: "#ffffff",

                  position: "absolute",

                  transform: "rotateY(180deg)", // Flip the back side

                  backfaceVisibility: "hidden", // Hide front when back is visible

                }}

              >

                <Typography

                  variant="h6"

                  sx={{

                    fontWeight: "bold",

                    fontSize: "1.5rem",

                    marginBottom: "10px",

                  }}

                >

                  {industry.title}

                </Typography>

                <Typography variant="body2">{industry.description}</Typography>

              </motion.div>

            </motion.div>

          </Box>

        ))}

      </Box>


      {/* Scroll Animation: Reveal text */}

      <motion.div

        ref={ref}

        initial="hidden"

        animate={controls}

        variants={revealVariants}

        style={{

          marginTop: "50px",

          textAlign: "center",

          color: "#b0b0b0", // Light gray text

          fontSize: "1.2rem",

        }}

      >

        Discover how we help businesses in these industries innovate and succeed.

      </motion.div>

    </Box>

  );

};


export default FlipIndustriesSection;
